alter:
1.add
2.drop
3.modify
4.rename

1.add:

syntax:

alter table tablename add columnname datatype;

example:

alter table student add email varchar(250);

2.drop:

syntax:

alter table tablename drop column columnname;

example:

alter table student drop column email;

3.modify:

syntax:

alter table tablename modify column columnname datatype;

example:

alter table student modify column phno bigint;

4.rename:

syntax:

alter table tablename rename column old_column_name to new_column_name;

example:

alter table student rename column phno to contactno;
